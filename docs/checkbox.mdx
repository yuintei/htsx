```tsx
  <Checkbox id="tag-frontend" name="tags" value="frontend" label="Frontend" />

  <Checkbox id="tag-infra" name="tags" value="infra" label="Infra" />

// Checkboxes with the same name submit one value per checked item.
// Validate "at least one" in server or client code.
// Do not add required to every item unless every item must be checked.
```

## Code

```tsx title="ui/checkbox.tsx"
import type { Child, JSX } from "hono/jsx";
import { c } from "./c";

export function Checkbox({
  id,
  label,
  invalid,
  error,
  required,
  class: custom,
  ...props
}: Omit<JSX.IntrinsicElements["input"], "type"> & {
  label?: Child;
  invalid?: boolean;
  error?: string;
}) {
  const isInvalid = invalid || Boolean(error);
  const errorId = id && error ? `${id}-error` : undefined;

  return (
    <div class="flex flex-col gap-1">
      <label class="flex w-fit cursor-pointer items-center gap-2 text-sm has-disabled:cursor-not-allowed has-disabled:opacity-50">
        <input
          id={id}
          type="checkbox"
          required={required}
          aria-invalid={isInvalid || undefined}
          aria-describedby={errorId}
          class={c("size-4", custom)}
          {...props}
        />
        {label ? (
          <span>
            {label}
            {required ? <span class="ml-1 text-destructive">*</span> : null}
          </span>
        ) : null}
      </label>
      {error ? (
        <p id={errorId} class="text-sm text-destructive">
          {error}
        </p>
      ) : null}
    </div>
  );
}
```
