## Code

```tsx title="ui/popover.tsx"
import type { JSX, Child } from "hono/jsx";
import { c } from "./c";

export function Popover({
  id,
  popover = "auto",
  class: custom,
  children,
  ...props
}: JSX.IntrinsicElements["div"] & {
  id: string;
  children?: Child;
}) {
  return (
    <div
      id={id}
      popover={popover}
      class={c(
        "m-auto rounded-md border border-border bg-card text-card-foreground shadow-md backdrop:bg-background/50",
        custom,
      )}
      {...props}
    >
      {children}
    </div>
  );
}
```
